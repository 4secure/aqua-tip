   - authorize() returns true
   - rules(): email required|string|email, password required|string

3. Create backend/app/Http/Controllers/Auth/RegisterController.php (single-action invokable):
   - __invoke(RegisterRequest $request)
   - Create user with Hash::make($request->password) for password
   - Auth::login($user) to start session
   - Return response()->json($user, 201)

4. Create backend/app/Http/Controllers/Auth/LoginController.php (single-action invokable):
   - __invoke(LoginRequest $request)
   - Use Auth::attempt($request->only('email', 'password'))
   - If fails: throw ValidationException::withMessages(['email' => ['The provided credentials are incorrect.']])
   - On success: $request->session()->regenerate()
   - Return response()->json(Auth::user())

5. Create backend/app/Http/Controllers/Auth/LogoutController.php (single-action invokable):
   - __invoke(Request $request)
   - Auth::guard('web')->logout()
   - $request->session()->invalidate()
   - $request->session()->regenerateToken()
   - Return response()->json(['message' => 'Logged out'])

6. Edit backend/routes/api.php:
   ```php
   use App\Http\Controllers\Auth\RegisterController;
   use App\Http\Controllers\Auth\LoginController;
   use App\Http\Controllers\Auth\LogoutController;
   use Illuminate\Http\Request;

   // Public auth routes
   Route::post('/register', RegisterController::class);
   Route::post('/login', LoginController::class);

   // Protected routes
   Route::middleware('auth:sanctum')->group(function () {
       Route::get('/user', function (Request $request) {
           return $request->user();
       });
       Route::post('/logout', LogoutController::class);
   });
   ```
