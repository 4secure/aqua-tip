The new config implements all 7 Nginx requirements (per D-01 through D-08 from CONTEXT.md):

```nginx
server {
    listen 8080;
    server_name _;
    root /var/www/public;
    index index.php;

    # INFRA-03 (D-05): Hide server version
    server_tokens off;

    # INFRA-06 (D-06): Cap request body at 2M (was 20M, no file uploads exist)
    client_max_body_size 2M;

    # INFRA-05 (D-01): HSTS -- Nginx only, not in SecurityHeaders middleware
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # INFRA-07 (D-02): Strict API-only CSP -- block all resource loading
    add_header Content-Security-Policy "default-src 'none'; frame-ancestors 'none'" always;

    # INFRA-01 (D-03): Block path traversal -- any URI containing ".." returns 403
    # Must appear before location / for correct Nginx processing order
    location ~* /\.\. {
        return 403;
    }

    # Main location
    location / {
        # INFRA-04 (D-07): Restrict HTTP methods
        if ($request_method !~ ^(GET|POST|PUT|DELETE|OPTIONS|PATCH)$) {
            return 405;
        }

        # CORS preflight -- MUST repeat security headers here due to add_header inheritance trap
        if ($request_method = 'OPTIONS') {
            add_header 'Access-Control-Allow-Origin' 'https://tip.aquasecure.ai' always;
            add_header 'Access-Control-Allow-Methods' 'GET, POST, PUT, DELETE, OPTIONS' always;
            add_header 'Access-Control-Allow-Headers' 'Content-Type, X-Requested-With, X-XSRF-TOKEN, Accept, Authorization' always;
            add_header 'Access-Control-Allow-Credentials' 'true' always;
            add_header 'Access-Control-Max-Age' '3600';
            # Repeat security headers (inheritance is lost in if blocks)
            add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
            add_header Content-Security-Policy "default-src 'none'; frame-ancestors 'none'" always;
            return 204;
        }

        try_files $uri $uri/ /index.php?$query_string;
    }

    # INFRA-02 (D-08): Only allow /index.php through FastCGI -- no arbitrary .php execution
    location = /index.php {
        fastcgi_pass 127.0.0.1:9000;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include fastcgi_params;
    }

    # Block hidden files (except .well-known)
    location ~ /\.(?!well-known).* {
        deny all;
    }
}
```

Key changes from current config:
1. Added `server_tokens off;` (INFRA-03)
2. Changed `client_max_body_size` from `20M` to `2M` (INFRA-06)
3. Added `Strict-Transport-Security` header at server level with `always` flag (INFRA-05)
4. Added `Content-Security-Policy` header at server level with `always` flag (INFRA-07)
5. Added `location ~* /\.\. { return 403; }` before `location /` (INFRA-01)
6. Added method restriction `if` block inside `location /` (INFRA-04)
7. Changed `location ~ \.php$` to `location = /index.php` (INFRA-02)
8. Repeated HSTS and CSP headers inside CORS OPTIONS `if` block to prevent add_header inheritance loss
