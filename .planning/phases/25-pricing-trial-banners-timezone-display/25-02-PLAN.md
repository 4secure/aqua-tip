    ```js
    import { useMemo } from 'react';
    import { useAuth } from '../contexts/AuthContext';

    export function useFormatDate() {
      const { timezone } = useAuth();

      return useMemo(() => {
        const dateFormatter = new Intl.DateTimeFormat('en-US', {
          year: 'numeric',
          month: 'short',
          day: 'numeric',
          timeZone: timezone,
        });

        const dateTimeFormatter = new Intl.DateTimeFormat('en-US', {
          year: 'numeric',
          month: 'short',
          day: 'numeric',
          hour: 'numeric',
          minute: '2-digit',
          timeZone: timezone,
        });

        const formatDate = (dateStr) => {
          if (!dateStr) return '--';
          try {
            return dateFormatter.format(new Date(dateStr));
          } catch {
            return String(dateStr);
          }
        };

        const formatDateTime = (dateStr) => {
          if (!dateStr) return '--';
          try {
            return dateTimeFormatter.format(new Date(dateStr));
          } catch {
            return String(dateStr);
          }
        };

        return { formatDate, formatDateTime };
      }, [timezone]);
    }
    ```
    - Per D-19/TZ-03: Unauthenticated users get UTC because AuthContext returns `timezone: user?.timezone ?? 'UTC'`
    - The hook uses native `Intl.DateTimeFormat` with the IANA timezone string -- no new dependencies (per D-16)
    - Output format matches D-17: "Mar 11, 2026" for date, "Mar 11, 2026 3:45 PM" for dateTime
